var array = [40,10,30,20];

// push() method
array.push(50);

console.log("array after pushing element 60 and 70 together",array);

array.pop();
console.log("array after removing the last element from the array",array)

array.shift();
console.log("array after removing the first element of an array",array)

// sort method

array.sort(function(a,b){return b-a});
console.log("array after sorting in descending order",array);
